<script>
  (function (_sr) {

    /*
     * Variables
     */


    const searchingPanel = document.querySelector('[data-widget="search"]');
    const filters = document.querySelector('#searchFilters');
    const searchTypesInputs = document.querySelectorAll('input[name="flatshare_type"]');
    const locationType = document.querySelector('input[name="location_type"]');
    const isLocationChange = document.querySelector('input[name="is_location_change"]');
    const searchInput = document.querySelector('input[name="search"]');
    let searchTypeInput;
      searchTypeInput = document.querySelector('input[name="flatshare_type"]:checked');
    const initialSearchType = searchTypeInput.value;
    const milesFromMax = document.querySelector('#miles_from_max');
    const searchFilterElements = document.querySelector('.search_filtered_by');
    const searchPanelDetails = document.querySelector('input[name="search"]');
    let filtersInitialState;
    const filtersReset = document.querySelector('[data-reset-link]');
    const availableSearch = (() => { try { return window.localStorage ? (localStorage.getItem('availableSearch') || '') : ''; } catch { return ''; } })();

    /*
     * Functions to replicate old jQuery functionality
     */

    const show = function (elem) {
      if (elem) {
        elem.style.display = 'block';
      }
    };

    const hide = function (elem) {
      if (elem) {
        elem.style.display = 'none';
      }
    };


    /*
     * Search Panel
     */

    const availSearchY = document.querySelector('[name="available_search"][value="Y"]');
    const availFrom = document.querySelector('input[name="available_from"]');

    if (availSearchY && availFrom) {
      availFrom.value = availableSearch;
    }

    searchTypesInputs.forEach((searchTypeInput) => {
      searchTypeInput.addEventListener('change', (e) => {
        const offeredInput = document.querySelector('input[value="offered"]');
        if ( offeredInput && offeredInput.checked ) {
          show(milesFromMax);
          show(searchFilterElements);
        } else {
          hide(milesFromMax);
          hide(searchFilterElements);
        }

        // Hide filters if user switches search type to something other
        // than the initial search
        if (initialSearchType === e.target.value) {
          show(filters);
        } else {
          hide(filters);
        }
      });
    });


    // LocationType to area when on change
    searchInput.addEventListener('input', function () {
      if (locationType) {
        locationType.value = 'area';
      }
      if (isLocationChange) {
        isLocationChange.value = '1';
      }
    });

    if (searchingPanel) {
      searchingPanel.querySelector('input[type="submit"]').addEventListener('click', function(){
        const flatshareType = document.querySelector('input[name="flatshare_type"]:checked');
        if(flatshareType && flatshareType.value !== initialSearchType) {
          filters.remove();
        }
      });
    }

    /*
     * Filters
     */

     // Ensure a radio button is always checked
    const selectRadios = function () {
      const radioGroups = {}

      if(filters) {
        filters.querySelectorAll('input[type="radio"]').forEach( function () {
          radioGroups[this.name] = true;
        });
      }

      for (name in radioGroups) {
        if (document.querySelectorAll('input[type="radio"][name="' + name + '"]:checked').length === 0) {
          filters.querySelector('input[name="' + name + '"][value=""]').checked = true;
        }
      }
    }

    // Initialise
    selectRadios();
    show(filtersReset);

    if (_sr.page.user_search) {
      filters.remove();
    }

    // Reset filters
    const defaultValues = function () {
      filters.querySelectorAll('input[type="number"]')
        .add('input[data-advanced-form]').val('');
      filters.querySelectorAll('input[type="checkbox"]')
        .add(filters.querySelectorAll('input[type="radio"]').checked = false);
      filters.querySelectorAll("option:selected").selected = false;
      document.querySelectorAll('#propertyType input').checked = true;
      selectRadios();
      document.querySelector('input[name="per"][value="pw"]').checked = true;
    };

    const applyFiltersSubmit = function() {
      const max_val = ~~ document.querySelector('input[name="max_rent"]').value;
      const min_val = ~~ document.querySelector('input[name="min_rent"]').value;

      if(min_val > max_val && max_val > 0) {
        document.querySelector('input[name="max_rent"]').value = min_val;
        document.querySelector('input[name="min_rent"]').value = max_val;
      }
      const availSearch = document.querySelector('input[name="available_search"]');
      if ( availSearch && !availSearch.checked ) {
        localStorage.removeItem('availableSearch');
      }
    }

    if (filters) {
      filtersInitialState = filters.innerHTML;
      if (filtersReset) {
        filtersReset.addEventListener('click', function (e) {
          e.preventDefault();
          defaultValues();
        });
      }
      const cancelLink = document.querySelector('[data-cancel-link]');
      if (cancelLink) {
        cancelLink.addEventListener('click', function(e){
          e.preventDefault();
          const filtersState = filters.innerHTML;
          if (filtersInitialState !== filtersState) {
            const searchFilters = document.querySelector("form[name='search-filter']");
            searchFilters.reset();
            selectRadios();
          }
          closeForm()
        });
      }
      const applyFiltersButton = filters.querySelector('input[type="submit"]');
      if (applyFiltersButton) {
        applyFiltersButton.addEventListener('click', function () {
          applyFiltersSubmit();
        });
      }
    }

  })(_sr);
</script>
